WINDOWS: icacls twmj_api.pem /reset; icacls twmj_api.pem /inheritance:r /grant:r "$($env:USERNAME):(R)"
MAC: chmod 400 "twmj_api.pem"

cd into instance

To check size: df -h

command = 'sudo vim /etc/nginx/sites-enabled/fastapi_nginx'

change ip to public IP of aws instance

press esc, and type :wq to save

sudo systemctl daemon-reload
sudo service nginx restart

activate env: source twmj/bin/activate

then can use uvicorn (run in background): nohup uvicorn twmj_api:twmj --host 0.0.0.0 --port 8000 > uvicorn.log 2>&1 &

Kill process: 
    ps aux | grep [u]vicorn
    kill <PID>
